    <script>
        // ================== CONFIG - UPDATE THESE AFTER DEPLOYING CONTRACT ==================
        let CONTRACT_ADDRESS = "0x0000000000000000000000000000000000000000"; // ← REPLACE with your deployed address
        const ABI = [
            "function publicMint(uint256 quantity) payable",
            "function whitelistMint(uint256 quantity, bytes32[] proof) payable",
            "function presaleMint(uint256 quantity) payable",
            "function totalSupply() view returns (uint256)",
            "function balanceOf(address) view returns (uint256)",
            "function ownerOf(uint256) view returns (address)",
            "function tokenURI(uint256) view returns (string)",
            "function mintPrice() view returns (uint256)",
            "function maxSupply() view returns (uint256)",
            "function currentPhase() view returns (uint8)",
            "function walletMints(address) view returns (uint256)",
            "function revealed() view returns (bool)"
        ];
        // ================================================================================

        let provider;
        let signer;
        let contract;
        let userAddress = null;
        let currentChainId = null;

        const CHAIN_CONFIG = {
            ethereum: { chainId: "0x1", rpc: "https://eth.llamarpc.com", name: "Ethereum", explorer: "https://etherscan.io" },
            base: { chainId: "0x2105", rpc: "https://mainnet.base.org", name: "Base", explorer: "https://basescan.org" },
            polygon: { chainId: "0x89", rpc: "https://polygon-rpc.com", name: "Polygon", explorer: "https://polygonscan.com" }
        };

        async function initTailwind() {
            tailwind.config = { content: ["./**/*.{html,js}"] };
        }

        async function connectWallet() {
            if (!window.ethereum) {
                alert("Please install MetaMask or another wallet");
                return;
            }
            try {
                provider = new ethers.providers.Web3Provider(window.ethereum);
                await provider.send("eth_requestAccounts", []);
                signer = provider.getSigner();
                userAddress = await signer.getAddress();
                
                document.getElementById("connectText").innerHTML = `✓ ${userAddress.slice(0,6)}...${userAddress.slice(-4)}`;
                document.getElementById("walletInfo").classList.remove("hidden");
                document.getElementById("walletInfo").innerHTML = `
                    <span class="font-mono text-emerald-400">${userAddress.slice(0,6)}...${userAddress.slice(-4)}</span>
                    <button onclick="disconnect()" class="text-xs px-3 py-1 bg-red-500/20 text-red-400 rounded-2xl">Disconnect</button>
                `;
                
                await loadContract();
                await refreshDashboard();
            } catch (e) {
                console.error(e);
                alert("Failed to connect wallet");
            }
        }

        function disconnect() {
            userAddress = null;
            document.getElementById("connectText").innerHTML = "Connect Wallet";
            document.getElementById("walletInfo").classList.add("hidden");
            location.reload();
        }

        async function loadContract() {
            if (!signer) return;
            contract = new ethers.Contract(CONTRACT_ADDRESS, ABI, signer);
            
            // Auto detect current phase
            try {
                const phase = await contract.currentPhase();
                const phaseNames = ["Inactive", "Whitelist", "Presale", "Public"];
                document.getElementById("phaseDisplay").innerHTML = `
                    <span class="px-4 py-1 bg-purple-500 text-black rounded-3xl text-xs font-bold">${phaseNames[phase]}</span>
                    <span class="ml-3 text-zinc-400">Phase active • Mint live</span>
                `;
            } catch(e) {}
        }

        async function refreshDashboard() {
            if (!contract || !userAddress) return;
            
            // Total supply
            try {
                const total = await contract.totalSupply();
                const max = await contract.maxSupply();
                document.getElementById("totalSupplyDisplay").textContent = `${total.toString()} / ${max.toString()}`;
                document.getElementById("mintedDisplay").textContent = total.toString();
            } catch(e) {}
            
            // User balance (NFTs owned)
            try {
                const balance = await contract.balanceOf(userAddress);
                document.getElementById("ownedNFTs").textContent = balance.toString();
                document.getElementById("userBalanceDisplay").textContent = balance.toString();
                document.getElementById("userAddress").textContent = `${userAddress.slice(0,8)}...`;
            } catch(e) {}
            
            // Mock recent tx (real apps would listen to events)
            document.getElementById("txHistory").innerHTML = `
                <div class="flex justify-between"><span class="text-emerald-400">Minted #142</span><span class="text-zinc-400">2m ago</span></div>
                <div class="flex justify-between"><span class="text-emerald-400">Minted #141</span><span class="text-zinc-400">1h ago</span></div>
            `;
        }

        async function mint() {
            if (!contract || !userAddress) {
                alert("Connect wallet first");
                return;
            }
            const qtyInput = document.getElementById("quantityInput");
            const quantity = parseInt(qtyInput.value);
            if (quantity < 1 || quantity > 5) {
                alert("Quantity must be 1-5");
                return;
            }
            
            const mintBtn = document.getElementById("mintBtn");
            const originalText = mintBtn.innerHTML;
            mintBtn.disabled = true;
            mintBtn.innerHTML = `<span class="animate-pulse">CONFIRMING IN WALLET...</span>`;
            
            try {
                const price = await contract.mintPrice();
                const value = price.mul(quantity);
                
                // Public mint (you can extend to detect phase and call whitelistMint/presaleMint if needed)
                const tx = await contract.publicMint(quantity, { value: value });
                console.log("Transaction sent:", tx.hash);
                
                // Wait for confirmation
                await tx.wait();
                
                alert(`✅ Successfully minted ${quantity} GrokPunks! Tx: ${tx.hash}`);
                await refreshDashboard();
            } catch (error) {
                console.error(error);
                if (error.code === 4001) {
                    alert("Transaction rejected by user");
                } else {
                    alert("Mint failed. Check console or gas.");
                }
            } finally {
                mintBtn.disabled = false;
                mintBtn.innerHTML = originalText;
            }
        }

        function showChainModal() {
            document.getElementById("chainModal").classList.remove("hidden");
            document.getElementById("chainModal").classList.add("flex");
        }

        function hideChainModal() {
            const modal = document.getElementById("chainModal");
            modal.classList.add("hidden");
            modal.classList.remove("flex");
        }

        async function switchChain(chainKey) {
            hideChainModal();
            const config = CHAIN_CONFIG[chainKey];
            
            try {
                await window.ethereum.request({
                    method: 'wallet_switchEthereumChain',
                    params: [{ chainId: config.chainId }]
                });
                currentChainId = config.chainId;
                document.getElementById("currentChain").innerHTML = config.name;
                
                // Re-init contract on new chain
                if (signer) await loadContract();
            } catch (switchError) {
                if (switchError.code === 4902) {
                    // Add chain if not added
                    await window.ethereum.request({
                        method: 'wallet_addEthereumChain',
                        params: [{
                            chainId: config.chainId,
                            chainName: config.name,
                            rpcUrls: [config.rpc],
                            blockExplorerUrls: [config.explorer]
                        }]
                    });
                } else {
                    console.error(switchError);
                }
            }
        }

        async function viewOnExplorer() {
            if (!currentChainId) {
                alert("Switch chain first");
                return;
            }
            const chainKey = Object.keys(CHAIN_CONFIG).find(k => CHAIN_CONFIG[k].chainId === currentChainId);
            if (chainKey && CONTRACT_ADDRESS !== "0x0000000000000000000000000000000000000000") {
                const explorer = CHAIN_CONFIG[chainKey].explorer;
                window.open(`${explorer}/address/${CONTRACT_ADDRESS}`, '_blank');
            } else {
                alert("Deploy contract and switch chain first");
            }
        }

        // Initialize everything
        window.onload = async () => {
            await initTailwind();
            // Make chain selector clickable
            document.getElementById("chainSelector").addEventListener("click", showChainModal);
            
            // Live gas/price updater (demo)
            setInterval(() => {
                if (document.getElementById("priceDisplay")) {
                    const qty = parseInt(document.getElementById("quantityInput").value) || 1;
                    document.getElementById("priceDisplay").textContent = `${(0.005 * qty).toFixed(3)} ETH`;
                    document.getElementById("gasEstimate").innerHTML = `Gas ~ $${(0.1 * qty).toFixed(2)} • Total ~ ${(0.005 * qty).toFixed(3)} ETH`;
                }
            }, 1500);
            
            console.log("%c✅ GrokPunks full minting platform ready! Deploy contract then update CONTRACT_ADDRESS in this file.", "color:#00ff9d; font-size:11px");
        };
    </script>
