pragma solidity ^0.8.24;

import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/utils/cryptography/MerkleProof.sol";
import "@openzeppelin/contracts/utils/Strings.sol";
import "erc721a/contracts/ERC721A.sol";
import "@openzeppelin/contracts/interfaces/IERC2981.sol";

contract GrokPunks is ERC721A, Ownable, Pausable, ReentrancyGuard, IERC2981 {
    using Strings for uint256;

    uint256 public maxSupply = 10000;
    uint256 public mintPrice = 0.005 ether;
    uint256 public maxPerWallet = 10;
    uint256 public maxPerTx = 5;
    bool public revealed = false;
    string public hiddenURI = "ipfs://QmHiddenMetadataHash/";
    string public baseURI = "";
    bytes32 public merkleRoot;
    uint256 public totalMinted;
    mapping(address => uint256) public walletMints;
    mapping(address => uint256) public whitelistMints;

    enum Phase { Inactive, Whitelist, Presale, Public }
    Phase public currentPhase = Phase.Inactive;

    constructor() ERC721A("GrokPunks", "GP") {
        _setDefaultRoyalty(msg.sender, 750); // 7.5% royalty
    }

    function publicMint(uint256 quantity) external payable whenNotPaused nonReentrant {
        require(currentPhase == Phase.Public, "Public mint not active");
        require(quantity > 0 && quantity <= maxPerTx, "Invalid quantity");
        require(totalSupply() + quantity <= maxSupply, "Exceeds max supply");
        require(msg.value >= mintPrice * quantity, "Insufficient funds");
        require(walletMints[msg.sender] + quantity <= maxPerWallet, "Exceeds wallet limit");

        walletMints[msg.sender] += quantity;
        totalMinted += quantity;
        _safeMint(msg.sender, quantity);
    }

    function whitelistMint(uint256 quantity, bytes32[] calldata proof) external payable whenNotPaused nonReentrant {
        require(currentPhase == Phase.Whitelist, "Whitelist not active");
        require(MerkleProof.verify(proof, merkleRoot, keccak256(abi.encodePacked(msg.sender))), "Not whitelisted");
        require(quantity > 0 && quantity <= maxPerTx, "Invalid quantity");
        require(totalSupply() + quantity <= maxSupply, "Exceeds max supply");
        require(msg.value >= mintPrice * quantity, "Insufficient funds");
        require(whitelistMints[msg.sender] + quantity <= 3, "Exceeds whitelist limit");

        whitelistMints[msg.sender] += quantity;
        walletMints[msg.sender] += quantity;
        totalMinted += quantity;
        _safeMint(msg.sender, quantity);
    }

    function presaleMint(uint256 quantity) external payable whenNotPaused nonReentrant {
        require(currentPhase == Phase.Presale, "Presale not active");
        require(quantity > 0 && quantity <= maxPerTx, "Invalid quantity");
        require(totalSupply() + quantity <= maxSupply, "Exceeds max supply");
        require(msg.value >= mintPrice * quantity, "Insufficient funds");
        require(walletMints[msg.sender] + quantity <= maxPerWallet, "Exceeds wallet limit");

        walletMints[msg.sender] += quantity;
        totalMinted += quantity;
        _safeMint(msg.sender, quantity);
    }

    function airdrop(address[] calldata recipients, uint256[] calldata quantities) external onlyOwner {
        require(recipients.length == quantities.length, "Length mismatch");
        for (uint256 i = 0; i < recipients.length; i++) {
            require(totalSupply() + quantities[i] <= maxSupply, "Exceeds supply");
            totalMinted += quantities[i];
            _safeMint(recipients[i], quantities[i]);
        }
    }

    function teamMint(uint256 quantity) external onlyOwner {
        require(totalSupply() + quantity <= maxSupply, "Exceeds supply");
        totalMinted += quantity;
        _safeMint(owner(), quantity);
    }

    function revealCollection() external onlyOwner { revealed = true; }
    function setPhase(Phase _phase) external onlyOwner { currentPhase = _phase; }
    function setMerkleRoot(bytes32 _merkleRoot) external onlyOwner { merkleRoot = _merkleRoot; }
    function setMintPrice(uint256 _price) external onlyOwner { mintPrice = _price; }
    function setMaxSupply(uint256 _supply) external onlyOwner { require(_supply >= totalSupply()); maxSupply = _supply; }
    function setBaseURI(string memory _uri) external onlyOwner { baseURI = _uri; }
    function setHiddenURI(string memory _uri) external onlyOwner { hiddenURI = _uri; }
    function pause() external onlyOwner { _pause(); }
    function unpause() external onlyOwner { _unpause(); }

    function withdraw() external onlyOwner {
        payable(owner()).transfer(address(this).balance);
    }

    function tokenURI(uint256 tokenId) public view override returns (string memory) {
        require(_exists(tokenId), "URI query for nonexistent token");
        if (!revealed) return string(abi.encodePacked(hiddenURI, tokenId.toString(), ".json"));
        return string(abi.encodePacked(baseURI, tokenId.toString(), ".json"));
    }

    function royaltyInfo(uint256, uint256 salePrice) public view override returns (address, uint256) {
        return (owner(), (salePrice * 750) / 10000);
    }

    function supportsInterface(bytes4 interfaceId) public view override(ERC721A, IERC165) returns (bool) {
        return interfaceId == type(IERC2981).interfaceId || super.supportsInterface(interfaceId);
    }
}

const hre = require("hardhat");

async function main() {
  const GrokPunks = await hre.ethers.getContractFactory("GrokPunks");
  const grokpunks = await GrokPunks.deploy();

  await grokpunks.deployed();
  console.log("GrokPunks deployed to:", grokpunks.address);

  // Optional: verify
  // await hre.run("verify:verify", { address: grokpunks.address });
}

main().catch((error) => {
  console.error(error);
  process.exitCode = 1;
});

// Generate metadata
function generateMetadata(tokenId) {
  return {
    name: `GrokPunks #${tokenId}`,
    description: "GrokPunks - A utility NFT collection built for long-term value and community.",
    image: `ipfs://YOUR_CID/${tokenId}.png`,
    attributes: [
      { trait_type: "Background", value: ["Neon", "Cyber", "Void"][Math.floor(Math.random()*3)] },
      { trait_type: "Eyes", value: ["Laser", "Classic", "Grok"][Math.floor(Math.random()*3)] },
      { trait_type: "Rarity", value: tokenId % 100 === 0 ? "Legendary" : "Common" }
    ]
  };
}

// Run this to generate all JSONs
for (let i = 1; i <= 10000; i++) {
  const metadata = generateMetadata(i);
  fs.writeFileSync(path.join(__dirname, 'metadata', `${i}.json`), JSON.stringify(metadata, null, 2));
}
